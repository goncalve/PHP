\begin{frame}[fragile]  {Redirection en PHP}
\begin{itemize}
	\item Se fait à l'aide de la fonction header('Location: chemin') qui envoie une redirection au navigateur. 
	
\end{itemize}
	
\lstset{language=php, commentstyle=\color{blue}\textit, backgroundcolor=\color{fondcode}, keywordstyle=\color{red}\bfseries, basicstyle=\ttfamily\small, showstringspaces=false, stringstyle=\color{violet}, breaklines=true, tabsize=1}
\begin{lstlisting}
	<?php 
    header('Location: index.php?status=right'); 
    exit(0); // ou exit(); ou exit;
	?>
\end{lstlisting}
\begin{itemize}
	\item Pas d'espace avant les deux points et un espace après
	\item Cette redirection doit être effectuée avant l'envoi de toute sortie au navigateur donc :
	\begin{itemize}
		\item aucun appel de la fonction echo avant notamment dans le PHP
		\item pas d'espace ou de retour-charriot dans le HTML 
	\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}  {Bufferisation de sortie}
\begin{itemize}
	\item \textbf{ob\string_start()} démarre la temporisation de sortie. Tant qu'elle est enclenchée, aucune donnée, hormis les en-têtes, n'est envoyée au navigateur, mais temporairement mise en tampon.
	\item \textbf{ob\string_get\string_clean()} lit le contenu courant du tampon de sortie puis l'efface. Retourne le contenu sous forme de chaîne de caractères.
	\item \textbf{ob\string_get\string_flush()} lit le contenu courant du tampon de sortie puis l'efface. Retourne le contenu sous forme de chaîne de caractères + stoppe la temporisation.
\end{itemize}
\end{frame}


\begin{frame}[fragile]  {Les sessions PHP (1/2)}
\lstset{language=php, commentstyle=\color{blue}\textit, backgroundcolor=\color{fondcode}, keywordstyle=\color{red}\bfseries, basicstyle=\ttfamily\small, showstringspaces=false, stringstyle=\color{violet}, breaklines=true, tabsize=1}
\begin{lstlisting}
<?php
  session_start(); 
  $_SESSION['views'] = 1; // stocke une variable de session
  echo "Pageviews = ". $_SESSION['views']; //accède à une variable de session
?>
\end{lstlisting}
\begin{itemize}
	\item session\_start() démarre le système de session et si le visiteur vient d'arriver un identifiant de session (PHPSESSID) est généré. Il sera transmis de page en page grâce à un cookie ou à défaut peut être transmis via l'URL.
	\item \$\_SESSION est un tableau de variables superglobales qui persiste sur le serveur tant que la session n'est pas terminée. Pour pouvoir y accéder, il faut démarrer le système de session dans la page.
	\item session\_start doit forcément être appelé en tout début de page.
\end{itemize}
\end{frame}

\begin{frame}[fragile]  {Les sessions PHP (2/2)}
\lstset{language=php, commentstyle=\color{blue}\textit, backgroundcolor=\color{fondcode}, keywordstyle=\color{red}\bfseries, basicstyle=\ttfamily\small, showstringspaces=false, stringstyle=\color{violet}, breaklines=true, tabsize=1}

<?php

	\item En pratique, les sessions sont automatiquement détruites au bout d'un certain temps d'inactivité ou si vous quittez votre navigateur.
\end{itemize}
\end{frame}

\begin{frame}[fragile]   {Les cookies}
\begin{itemize}
	\item Ce sont des informations qu'on peut stockées temporairement dans un fichier sur le poste du client
	\item Création d'un cookie : set cookie (name, value, expire, path, domain, secure, httpOnly) 
		\begin{itemize}
			\item expire est une date d'expiration exprimée en nombre de secondes depuis le 1er janvier 1970. C'est l'unité utilisée par la fonction time d'UNIX
			\item secure rend le cookie inaccessible si la connexion n'est pas sécurisée (HTTPS)
			\item httpOnly rend le cookie inaccessible à d'autres protocoles que le HTTP
		\end{itemize}
	\item Le cookie n'est accessible qu'au prochain chargement d'une page.
\end{itemize}
\lstset{language=php, commentstyle=\color{blue}\textit, backgroundcolor=\color{fondcode}, keywordstyle=\color{red}\bfseries, basicstyle=\ttfamily\small, showstringspaces=false, stringstyle=\color{violet}, breaklines=true, tabsize=1}
\begin{lstlisting}
<?php
  $value = 'toto';
   // Création du cookie qui expirera dans une heure accessible en HTTP seulement
  setcookie("TestCookie", $value, time()+3600, null, null, false, true);
?>

<?php
 //Destruction du cookie
 setcookie ("TestCookie", "", time() - 3600,  null, null, false, true);
?>
\end{lstlisting}	


\end{frame}

\begin{frame}   {Les fichiers (1/2)}
\begin{itemize}
	\item \$fi = fopen(nom\_fichier, mode) : ouverture d'un fichier (possibilité de fournir  une URL HTTP ou FTP pour les fichiers distants). Retourne une ressource représentant le pointeur de fichier, ou FALSE si une erreur survient.
	\item fclose() : fermeture du fichier
	\item fgets (\$fi) Retourne une string, la ligne courante sur laquelle se trouve le pointeur du fichier.\\
	fgetss permet de lire une ligne dans un fichier en supprimant les balises HTML
	\item rtrim(fgets (\$fi)) Idem mais on enlève le retour chariot.
\end{itemize}

\end{frame}

	\item feof(\$fi) Retourne TRUE si le pointeur est à la fin du fichier.
	\item rewind(\$fi), fseek(\$fi, position) déplace le pointeur dans le fichier.
	\item et encore beaucoup d'autres fonctions pour les fichiers ...
	\item et d'autres pour manipuler les répertoires ... 
\end{itemize}
\end{frame}

\begin{frame}    {Le tableau de superglobales \$\_FILES}
\begin{itemize}
	\item  \$\_FILES['mon\_fichier']['name'] contient le nom du fichier
	\item  \$\_FILES['mon\_fichier']['type'] contient le type du fichier
	\item  \$\_FILES['mon\_fichier']['size'] contient sa taille en octets
	\item  \$\_FILES['mon\_fichier']['tmp\_name'] contient l'emplacement temporaire du fichier
	\item  \$\_FILES['mon\_fichier']['error'] contient le code d'erreur (0 si pas d'erreur; 1 et 2 trop gros; ...)
\end{itemize}
\begin{itemize}
	\item move\_uploaded\_file(\$\_FILES['mon\_fichier']['tmp\_name'], \$nouveau\_nom) permet de déplacer un fichier téléchargé par PHP

\end{itemize}
\end{frame}
